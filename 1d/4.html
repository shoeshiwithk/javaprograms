4. Program to print the duplicate elements of an array.
    class Func{
        void Frequency(int arr[]){
        int n=arr.length;
        boolean visited[]=new boolean[n];
        for(int i=0;i<n;i++){
        if (visited[i]==true)
        continue;
        
        int count=1;
        for(int j=i+1;j<n;j++){
        if (arr[i]==arr[j]){
        visited[j]=true;
        count++;
        }
        }
        if(count>1)
        System.out.println(arr[i]+" frequency = "+count);
        }
        
        }
        }
        
        class main{
        public static void main(String st[]){
        int arr[]=new int[]{1,1,1,4,4,56,7};
        Func obj=new Func();
        System.out.println("The duplicate elements are:");
        obj.Frequency(arr);
        }
        }
