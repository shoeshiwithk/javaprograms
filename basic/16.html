14. Write a Java program to print the even numbers from 1 to 20 

import java.util.ArrayList;
class Func{
ArrayList<Integer> EvenNum(){
ArrayList<Integer> arr= new ArrayList<>();
for(int i=1;i<21;i++){
if(i%2==0){
arr.add(i);
}
}
return arr;
}
}

class A{
public static void main(String st[]){
Func obj=new Func();
ArrayList<Integer> arr=obj.EvenNum();
for(int i: arr){
System.out.println(i);
}
}
}
