17. Write a Java program to compute the sum of the first 100 prime numbers 

import java.util.ArrayList;
class Func{
int SumOfPrimeNum(){
ArrayList<Integer> list= new ArrayList<>();
list.add(2);
int count=1;
for(int j=3;count<100;j++){
boolean isPrime=true;
for(int i=2; i<j; i++){
if(j%i==0){
isPrime=false;
break;
}
}
if(isPrime){
list.add(j);
count++;
}
}


int size=list.size();
int sum=0;

for(int i=0;i<size;i++){
sum=sum+list.get(i);
System.out.print(list.get(i)+" ");
}

return sum;
}
}

class A{
public static void main(String st[]){
Func obj=new Func();
System.out.println("\nSum of first 100 Prime Numbers is :
"+obj.SumOfPrimeNum());
}
}
