26. Java Program to find area of equilateral triangle 

import java.lang.Math;
class Func{
float EquiTriArea(float a){
float x=(float)Math.sqrt(3);
float area=(x/4)*a*a;
return area;
}
}

class A{
public static void main(String st[]){
Func obj = new Func();
float a=5;
System.out.println("The area of Equilateral triangle with side "+a+" cm
is: "+obj.EquiTriArea(a));
}
}
