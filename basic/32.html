30. Java Program to find volume of sphere 

    class Func{
        static final float PI=22f/7f;
        float SphereVol(float r){
        float area = (4f/3f)*PI*r*r*r;
        return area;
        }
        }
        
        class A{
        public static void  main(String st[]){
        Func obj = new Func();
        float r=7;
        System.out.println("Volume of sphere with radius "+r+" units 
        is: "+obj.SphereVol(r));
        }
        }
