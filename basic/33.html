31. Java Program to find volume of cylinder 

    class Func{
        static final float PI=22f/7f;
        float CylicVol(float r, float h){
        float area = PI*r*r*h;
        return area;
        }
        }
        
        class A{
        public static void  main(String st[]){
        Func obj = new Func();
        float r=7, h=7;
        System.out.println("Volume of sphere with radius "+r+" and 
        height "+h+"  units is: "+obj.CylicVol(r,h));
        }
        }
