33. Java Program to find volume of cone 

    class Func{
        static final float PI=22f/7f;
        float ConeVol(float r, float h){
        float area = (PI*r*r*h)*(1f/3f);
        return area;
        }
        }
        
        class A{
        public static void  main(String st[]){
        Func obj = new Func();
        float r=7, h=7;
        System.out.println("Volume of cone with radius "+r+" and height 
        "+h+"  units is: "+obj.ConeVol(r,h));
        }
        }
