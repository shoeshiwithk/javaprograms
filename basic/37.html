35. Java program to find surface area of cylinder 

    class Func{
        static final float PI=22f/7f;
        float CylicSA(float r, float h){
        float area = (2*PI*r*h)+(2*PI*r*r);
        return area;
        }
        }
        
        class A{
        public static void  main(String st[]){
        Func obj = new Func();
        float r=7f, h=7f;
        System.out.println("Surface area of cylinder with radius "+r+" 
        and height "+h+"  units is: "+obj.CylicSA(r,h));
        }
        }
