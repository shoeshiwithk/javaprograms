37. Java program to calculate average marks 

    class Func{
        float AvgMrk(int arr[]){
        int size=arr.length;
        float avg=0;
        for (int i=0;i<size;i++){
        avg=avg+arr[i];
        }
        avg=avg/size;
        return avg;
        }
        }
        
        class A{
        public static void main(String st[]){
        Func obj=new Func();
        int arr[]=new int[]{65,65,65,65,65};
        System.out.println("Average marks are "+obj.AvgMrk(arr));
        
        }
        }
