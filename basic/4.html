4. Write a Java program to print the sum, multiply, subtract, divide and remainder of two numbers 

    class Func{
        double product(double a, double b){
        double c=a*b;
        return c;
        }
        
        double sum(double a, double b){
        double c=a+b;
        return c;
        }
        
        double sub(double a, double b){
        double c=a-b;
        return c;
        }
        
        double divide(double a, double b){
        double c=a/b;
        return c;
        }
        
        double rem(double a, double b){
        double c=a%b;
        return c;
        }
        }
        
        class A{
        public static void main(String st[]){
        Func obj=new Func();
        System.out.println(obj.sum(6,3));
        System.out.println(obj.product(6,3));
        System.out.println(obj.sub(6,3));
        System.out.println(obj.divide(6,3));
        System.out.println(obj.rem(6,3));
        }
        }
