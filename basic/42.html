40. Java program to find factorial of any number 

    class Func{
        int Factorial(int n){
        int fact=1;
        if(n==0)
        return 1;
        else{
        for(int i=2;i<=n;i++){
        fact=fact*i;
        }
        return fact;
        }
        }
        }
        
        class A{
        public static void main(String st[]){
        Func obj=new Func();
        int n=0;
        System.out.println("Factorial of "+n+" is "+obj.Factorial(n));
        }
        }
