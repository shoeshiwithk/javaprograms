    class Func{
        float MonthBill(float cr, float pr, float rpu){
        // pr=previous reading; cr=current reading; rpu=rate per unit; 
        rom=reading of month;
        float rom=cr-pr;
        float bill=rom*rpu;
        return bill;
        }
        }
