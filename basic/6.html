6. Write a Java program to swap two variables 

    class Func{
        void swap(int a, int b){
        System.out.println("Before swap:\nA = "+a+" and B = "+b);
        int temp=a;
        a=b;
        b=temp;
        System.out.println("After swap:\nA = "+a+" and B = "+b);
        }
        }
        
        class A{
        public static void main(String at[]){
        Func obj=new Func();
        obj.swap(4,5);
        }
        }
