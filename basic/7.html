7. Write a Java program to convert a decimal number to binary numbers 

    class Func{
        int[] binary(int a){
        int bits = Integer.SIZE-Integer.numberOfLeadingZeros(a);
        int arr[]=new int[bits];
        for (int i=bits-1;a!=0;i--){
        arr[i]=a%2;
        a=a/2;
        }
        return arr;
        }
        }
        
        class A{
        public static void main(String st[]){
        Func obj=new Func();
        int num =500;
        int arr[]=obj.binary(num);
        int size=arr.length;
        System.out.print("Binary number of "+num+" is: ");
        for(int i=0;i<size;i++){
        System.out.print(arr[i]);
        }
        }
        }
