8. Write a Java program to convert a binary number to decimal number 

    class Func{
        int decimal(int a){
        
        int count=0, dec=0;
        for(int temp=a;temp!=0;temp/=10,count++);
        
        for(int i=0;i<count;i++){
        int x = (int)Math.pow(2,i);
        dec=((a%10)*x)+dec;
        a/=10;
        }
        return dec;
        }
        }
